<script>
const menu = document.getElementById("menu");
const layout = document.getElementById("layout");
const button = document.getElementById("toggleMenu");

button.addEventListener("click", () => {
    menu.classList.toggle("vertical");
    layout.classList.toggle("vertical");
    menu.querySelectorAll(".open").forEach(el => el.classList.remove("open"));
});

// accordion w menu pionowym
menu.addEventListener("click", e => {
    const link = e.target.closest("a");
    if(!link) return;
    const li = link.parentElement;
    if(!li.classList.contains("has-children")) return;
    if(menu.classList.contains("vertical")) {
        e.preventDefault();
        li.classList.toggle("open");
    }
});
</script>
